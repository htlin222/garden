    def nussinov_rna_folding(rna_seq):
    n = len(rna_seq)
    dp = [[0] * n for _ in range(n)]

    # å¡«å……å‹•æ…‹è¦åŠƒçŸ©é™£
    for length in range(2, n):
        for i in range(n - length):
            j = i + length
            dp[i][j] = max(dp[i + 1][j],
                           dp[i][j - 1],
                           dp[i + 1][j - 1] + (rna_seq[i] == rna_seq[j]),
                           max(dp[i][k] + dp[k + 1][j] for k in range(i, j)))

    return dp[0][n - 1]
