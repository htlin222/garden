#!/bin/bash

while span> $; do
  key="$1"

  case $key in
    --file)
      FILE="$2"
      shift
      ;;
    --token)
      TOKEN="$2"
      shift
      ;;
    *)
      ;;
  esac
  shift
done

CONTENT=$(cat $FILE)

curl -H "Authorization: token $TOKEN" \
  -H "Content-Type: application/json" \
  -X POST -d "{\"public\":true,\"files\":{\"$FILE\":{\"content\":\"$CONTENT\"}}}" \
  https://api.github.com/gists
